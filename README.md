    function json_load( url, type ){
      var json = null;
      $.ajax({
          'type':"GET",
          'async': false,
          'global': false,
          'cache': false,
          'url': url,
          'dataType': type,
          'success': function (data) {
              json = data;
          }
      });
      if( type == "csv" ){
        var json = json.split(/
      /g);
      }

      return json;
    }
